C++ Integration
Drop-in SDK for your application
Quick Setup

1. Copy vaultauth.hpp into your project.
2. Set VAULTAUTH_URL to your server URL.
3. Call VaultAuth::activate(key, hwid, username) on startup.
4. Call VaultAuth::validate() periodically as a heartbeat.

Usage Example
vaultauth.hpp
#include "vaultauth.hpp"

int main() {
    // Configure your server endpoint
    VaultAuth::setServerUrl("https://your-server.com");

    // Collect key + username from user input
    std::string key      = "VAULT-ABCD12-EF3456-...";
    std::string username = "john_doe";
    std::string hwid     = VaultAuth::getHWID(); // auto-generated

    // Activate (first run)
    VaultAuth::AuthResult result = VaultAuth::activate(key, hwid, username, "1.0.0");

    if (!result.success) {
        std::cerr << "Activation failed: " << result.message << std::endl;
        return 1;
    }

    std::cout << "Welcome, " << result.username << "! Plan: " << result.plan << std::endl;

    // Save session for heartbeat validation
    VaultAuth::saveSession(result);

    // ... your application loop ...

    // Heartbeat check (call every few minutes)
    if (!VaultAuth::validate(key, hwid).success) {
        // License revoked or expired — shut down
        exit(1);
    }

    // On logout / clean deactivate:
    VaultAuth::deactivate(key, hwid);

    return 0;
}
